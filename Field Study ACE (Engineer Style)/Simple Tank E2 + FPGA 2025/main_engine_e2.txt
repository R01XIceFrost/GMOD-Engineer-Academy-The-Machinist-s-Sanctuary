#September 24 - 28 2025

Owner = owner()

# settings
MAGNITUDE = 0.5
Direction = 0
MINIMUM = -10
MAX = 100
Throttle = 0

#-----------------

event keyPressed(Player:entity, Key:string, Down:number, KeyBind:string) { #I used KeyBind for universal keys (in case some weirdo out there uses something other than W & S) instead of Key:string
    if (Player != Owner) { exit() } #Reject everyone who isn't owner
    if (Player:inVehicle()) { #condition #2 checked if player is sitting on the chair
        #Forward & Backward Logic
        if (KeyBind == "forward") { 
            Direction = Down # Since Down is either 1 or 0, we can use this trick
            } elseif (KeyBind == "back") {
            Direction = -Down
        }
        #---------------------------------------------------------------------------------
        
        #Turning Logic for the Thruster (-1 or 1 or depending location of your Thruster!)
        if (KeyBind == "moveleft") {
            Result = -Down # -1 (Modify this code -Down to Down if it's not turning the right direction)
            } elseif (KeyBind == "moveright") {
            Result = Down # 1 (Modify this code Down to -Down if it's not turning the right direction)
        }         
    }
}
            
event tick() {
    
Throttle += Direction * MAGNITUDE #Ask Marquis Denneisk (one of developer) from Wiremod Discord how this line of code works (my head hurts...)
Turn = Result 
Throttle = clamp(Throttle, MINIMUM, MAX) #Clamps the Speed -10 to 100 according to the settings

    if (Throttle < 0) { #Activate the back thruster(s) if speed is below 0! NOTE: One thruster is not enough to turn the wheel I just realized during phrase finalizing docs -July 21, 2026
        Reverse = -1 # (Modify this code -1 to 1 if it's not reversing the back of baseplate direction properly)
     } else {
        Reverse = 0
    }   
 
Auto_Start = Owner:inVehicle()

if (Auto_Start) { #Activates the Fuel Tank when owner is sitting
    
    Fuel = 1
    
    if (!timerExists("Engine")) { #Activates the Engine next
        timer("Engine", 1, 0, function() { #Delay 1 sec, infinite
            Engine = 1            
            })
        }
    } else { #Stop Auto start activation when owner is not sitting
        stoptimer("Engine")
        Fuel = 0
        Engine = 0        
    }
 
#Wirelink stuff    

Fuel_WL["Active", number] = Fuel
Engine_WL["Active", number] = Engine
Engine_WL["Throttle", number] = Throttle
     
}
