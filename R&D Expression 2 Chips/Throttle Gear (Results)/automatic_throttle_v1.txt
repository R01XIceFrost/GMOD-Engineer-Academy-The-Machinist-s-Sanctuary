# settings
A = 0
MINIMUM = -10
MAX = 100
MAGNITUDE = 0.5
Direction = 0
#-----------------

event keyPressed(Player:entity, Key:string, Down:number, KeyBind:string) {
    if (Player:inVehicle() & Key == "w") { #Activates when player is in the chair and holding key w
        Direction = Down # Since Down is either 1 or 0, we can use this trick
    } elseif (Player:inVehicle() & Key == "s") { #Activates when player is in the chair and holding key s
        Direction = -Down
    }
}

event tick() {
    A += Direction * MAGNITUDE
    B = clamp(A, MINIMUM, MAX) #Clamps the Speed -10 to 100 
    Screen["A", number] = B #Plug & Play! Wire E2 Chip [WIRELINK] to Screen 
}
