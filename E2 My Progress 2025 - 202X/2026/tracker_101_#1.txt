event tick() {
    
if (Target:isValid()) { #if Target is found then do the following below!
    
PosTarget = Target:pos() #Target Current Pos in VECTOR

E2Pos = entity():pos() #E2 Chip Current Pos in VECTOR

Aim = (PosTarget - E2Pos):toAngle() #Converts VECTOR to ANGLES

E2Entity = entity() #Your E2 model as a turret

Pitch = Aim:pitch() #Gets Pitch from the Target
Yaw = Aim:yaw()     #Gets Yaw from the Target
Roll = Aim:roll()   #Gets Roll from the Target

OffsetAng = ang(Pitch - 30 ,Yaw - 5, Roll) #OffsetAng stuff
E2Entity:setAng(OffsetAng) #Set the Turret Offset angles according to the target's angles! 
#Or use default Aim angles

print(OffsetAng) #Print the results aka my debugging tool

    }
    
}
